Angular wrapper for Jed

See the Github page for the complete documentation

The module provides a filter and a directive for all kinds of translations.

Translations in controllers

{{ title }}

// controller
app.controller('HomeController', [
  '$scope', 'i18n', function($scope, i18n) {
    // load a page translation
    i18n.loadPage('home').then(function() {
      $scope.title = i18n._('{{ username }} is the king of the pop',
                             {username: $scope.username});
    });
]);
<!-- template -->
<p>{{ title }}</p>

Trans directive - Works as a tag or attribute

<input type="text" ng-model="nbr">
<input type="text" ng-model="username">
<trans
    count="nbr"
    when="{
        one: '{{ username }} has {{ nbr|number }} apple',
        plural: '{{ username }} has {{ nbr|number }} apples',
        none: '{{ username }} doesn\'t have any apples'
    }">
</trans>
<h4 trans
    count="nbr"
    when="{
        one: '{{ username }} has {{ nbr|number }} apple',
        plural: '{{ username }} has {{ nbr|number }} apples',
        none: '{{ username }} doesn\'t have any apples'
    }">
</h4>
